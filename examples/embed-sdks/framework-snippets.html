Framework install snippets

Copy into your host app. Replace emb_YOUR_KEY and Monti base URL. Allowlist this origin in tenant Embed settings.

Vue 3

npm install @monti/embed-vue @monti/embed-core

<script setup>
import { MontiEmbedVue } from '@monti/embed-vue'
</script>
<template>
  <MontiEmbedVue embed-key="emb_YOUR_KEY" api-base="http://localhost:8091" />
</template>

React

npm install @monti/embed-react @monti/embed-core

import { MontiEmbedReact } from '@monti/embed-react'

export function Widget() {
  return (
    <MontiEmbedReact embedKey="emb_YOUR_KEY" apiBase="http://localhost:8091" />
  )
}

Svelte

npm install @monti/embed-svelte @monti/embed-core

<script>
  import MontiEmbed from '@monti/embed-svelte/MontiEmbed.svelte'
</script>
<MontiEmbed embedKey="emb_YOUR_KEY" apiBase="http://localhost:8091" />

Web Component

npm install @monti/embed-web-component @monti/embed-core

<script type="module">
  import '@monti/embed-web-component'
</script>
<monti-embed embed-key="emb_YOUR_KEY" api-base="http://localhost:8091"></monti-embed>

Vanilla script (unchanged)

<script
  src="http://localhost:8091/embed/monti-embed.js"
  data-embed-key="emb_YOUR_KEY"
  data-position="bottom-right"
  async
></script>
